278 : child_(std::move(child)), maxRepeats_(maxRepeats) {}
279
280 BTStatus Tick(BTContext& context) override {
281 auto status = child_->Tick(context);
282 if (status == BTStatus::Running) {
283 return BTStatus::Running;
284 }
285
286 // Child finished (success or failure). Count the iteration.
287 ++currentCount_;
288 child_->Reset();
289
290 if (maxRepeats_ > 0 && currentCount_ >= maxRepeats_) {
291 currentCount_ = 0;
292 return BTStatus::Success;
293 }
294
295 return BTStatus::Running;
296 }

299 currentCount_ = 0;
300 child_->Reset();
301 }
