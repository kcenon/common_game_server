124 void Remove(Entity entity) override {
125 if (!Has(entity)) {

127 }

129 auto idx = sparse_[entity.id()];
130 auto lastIdx = static_cast<uint32_t>(dense_.size() - 1);

132 if (idx != lastIdx) {
133 // Swap the removed element with the last element.
134 dense_[idx] = std::move(dense_[lastIdx]);
135 entities_[idx] = entities_[lastIdx];

142 dense_.pop_back();
143 entities_.pop_back();
144 versions_.pop_back();
145 sparse_[entity.id()] = kInvalidIndex;

148 }
