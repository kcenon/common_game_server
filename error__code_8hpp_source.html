169constexpr std::string_view errorSubsystem(ErrorCode code) {
170 auto value = static_cast<uint32_t>(code);
171 auto category = value & 0xFF00;
172 switch (category) {
173 case 0x0000:
174 return "General";
175 case 0x0100:
176 return "Network";
177 case 0x0200:
178 return "Database";
179 case 0x0300:
180 return "ECS";
181 case 0x0400:
182 return "Plugin";
183 case 0x0500:
184 return "Auth";
185 case 0x0600:
186 return "Config";
187 case 0x0700:
188 return "Thread";
189 case 0x0800:
190 return "Logger";
191 case 0x0900:
192 return "Monitoring";
193 case 0x0A00:
194 return "Serialization";
195 case 0x0B00:
196 return "GameServer";
197 case 0x0C00:
198 return "Lobby";
199 case 0x0D00:
200 return "DBProxy";
201 case 0x0E00:
202 return "Gateway";
203 case 0x0F00:
204 return "Persistence";
205 default:
206 return "Unknown";
207 }
