113struct BinaryReader {
114 std::span<const uint8_t> data;
115 std::size_t pos = 0;
116
117 [[nodiscard]] bool canRead(std::size_t n) const { return pos + n <= data.size(); }
118
119 template <typename T>
120 bool readPrimitive(T& val) {
121 if (!canRead(sizeof(T)))
122 return false;
123 std::memcpy(&val, data.data() + pos, sizeof(T));
124 pos += sizeof(T);
125 return true;
126 }
127
128 bool readString(std::string& val) {
129 uint32_t len = 0;
130 if (!readPrimitive(len))
131 return false;
132 if (!canRead(len))
133 return false;
134 val.assign(reinterpret_cast<const char*>(data.data() + pos), len);
135 pos += len;
136 return true;
137 }
138};

140template <typename T>
141bool readBinaryField(BinaryReader& reader, T& val) {
142 if constexpr (std::is_same_v<T, bool>) {
143 uint8_t b = 0;

145 return false;
146 val = (b != 0);
147 return true;

149 return reader.readString(val);
150 } else if constexpr (std::is_arithmetic_v<T>) {
151 return reader.readPrimitive(val);
152 }
153 return false;

210struct JsonReader {
211 std::string_view data;
212 std::size_t pos = 0;
213
214 void skipWhitespace() {
215 while (pos < data.size() &&
216 (data[pos] == ' ' || data[pos] == '\t' || data[pos] == '\n' || data[pos] == '\r')) {
217 ++pos;
218 }
219 }
220
221 bool expect(char c) {
222 skipWhitespace();
223 if (pos < data.size() && data[pos] == c) {
224 ++pos;
225 return true;
226 }
227 return false;
228 }
229
230 bool readQuotedString(std::string& out) {
231 skipWhitespace();
232 if (pos >= data.size() || data[pos] != '"')
233 return false;
234 ++pos;
235 out.clear();
236 while (pos < data.size() && data[pos] != '"') {
237 if (data[pos] == '\\') {
238 ++pos;
239 if (pos >= data.size())
240 return false;
241 switch (data[pos]) {
242 case '"':
243 out += '"';
244 break;
245 case '\\':
246 out += '\\';
247 break;
248 case 'b':
249 out += '\b';
250 break;
251 case 'f':
252 out += '\f';
253 break;
254 case 'n':
255 out += '\n';
256 break;
257 case 'r':
258 out += '\r';
259 break;
260 case 't':
261 out += '\t';
262 break;
263 default:
264 out += data[pos];
265 break;
266 }
267 } else {
268 out += data[pos];
269 }
270 ++pos;
271 }
272 if (pos >= data.size())
273 return false;
274 ++pos; // skip closing quote
275 return true;
276 }
277
280 bool readRawValue(std::string& out) {
281 skipWhitespace();
282 if (pos >= data.size())
283 return false;
284 if (data[pos] == '"') {
285 return readQuotedString(out);
286 }
287 std::size_t start = pos;
288 while (pos < data.size() && data[pos] != ',' && data[pos] != '}' && data[pos] != ' ' &&
289 data[pos] != '\t' && data[pos] != '\n' && data[pos] != '\r') {
290 ++pos;
291 }
292 out = std::string(data.substr(start, pos - start));
293 return !out.empty();
294 }
295
297 void skipValue() {
298 skipWhitespace();
299 if (pos >= data.size())
300 return;
301 if (data[pos] == '"') {
302 std::string dummy;
303 readQuotedString(dummy);
304 } else {
305 while (pos < data.size() && data[pos] != ',' && data[pos] != '}') {
306 ++pos;
307 }
308 }
309 }
310};

312template <typename T>
313bool parseJsonValue(const std::string& raw, T& val) {
314 if constexpr (std::is_same_v<T, bool>) {
315 if (raw == "true") {
316 val = true;
317 return true;
318 }
319 if (raw == "false") {
320 val = false;
321 return true;
322 }
323 return false;
324 } else if constexpr (std::is_same_v<T, std::string>) {
325 val = raw;
326 return true;
327 } else if constexpr (std::is_floating_point_v<T>) {
328 try {
329 if constexpr (std::is_same_v<T, float>) {
330 val = std::stof(raw);
331 } else {
332 val = std::stod(raw);
333 }
334 return true;
335 } catch (...) {
336 return false;
337 }
338 } else if constexpr (std::is_integral_v<T>) {
339 try {
340 if constexpr (std::is_signed_v<T>) {
341 val = static_cast<T>(std::stoll(raw));
342 } else {
343 val = static_cast<T>(std::stoull(raw));
344 }
345 return true;
346 } catch (...) {
347 return false;
348 }
349 }
350 return false;
351}

400 template <typename T>
401 [[nodiscard]] std::vector<uint8_t> serializeBinary(const T& obj) const {
402 static_assert(detail::is_serializable_v<T>,
403 "Type must be registered with CGS_SERIALIZABLE");
404
405 std::vector<uint8_t> buf;
406 buf.reserve(128);
407
408 // Header: magic + version + field count
409 detail::writeBytes(buf, kBinaryMagic, 4);
410
411 uint32_t version = SerializableTraits<T>::schema_version;
412 detail::writePrimitive(buf, version);
413
414 auto fields = SerializableTraits<T>::fields();
415 constexpr auto numFields = static_cast<uint32_t>(std::tuple_size_v<decltype(fields)>);
416 detail::writePrimitive(buf, numFields);
417
418 // Fields in declaration order
419 detail::forEachField(fields, [&](const auto& fd, std::size_t) {
420 detail::writeBinaryField(buf, obj.*(fd.pointer));
421 });
422
423 return buf;
424 }

428 template <typename T>
429 [[nodiscard]] GameResult<T> deserializeBinary(std::span<const uint8_t> data) const {
430 static_assert(detail::is_serializable_v<T>,
431 "Type must be registered with CGS_SERIALIZABLE");
432
433 detail::BinaryReader reader{data};
434
435 // Verify magic bytes
436 uint8_t magic[4]{};
437 for (auto& m : magic) {
438 if (!reader.readPrimitive(m)) {
439 return GameResult<T>::err(
440 GameError(ErrorCode::InvalidBinaryData, "truncated binary header"));
441 }
442 }
443 if (std::memcmp(magic, kBinaryMagic, 4) != 0) {
444 return GameResult<T>::err(
445 GameError(ErrorCode::InvalidBinaryData, "invalid magic bytes"));
446 }
447
448 // Schema version (stored for forward compatibility)
449 uint32_t version = 0;
450 if (!reader.readPrimitive(version)) {
451 return GameResult<T>::err(
452 GameError(ErrorCode::InvalidBinaryData, "truncated version field"));
453 }
454
455 // Field count from the serialized data
456 uint32_t storedFieldCount = 0;
457 if (!reader.readPrimitive(storedFieldCount)) {
458 return GameResult<T>::err(
459 GameError(ErrorCode::InvalidBinaryData, "truncated field count"));
460 }
461
462 T obj{};
463 auto fields = SerializableTraits<T>::fields();
464 constexpr uint32_t currentFieldCount =
465 static_cast<uint32_t>(std::tuple_size_v<decltype(fields)>);
466
467 // Read min(stored, current) fields; extra fields keep defaults
468 uint32_t fieldsToRead = std::min(storedFieldCount, currentFieldCount);
469
470 detail::forEachField(fields, [&](const auto& fd, std::size_t idx) {
471 if (static_cast<uint32_t>(idx) < fieldsToRead) {
472 detail::readBinaryField(reader, obj.*(fd.pointer));
473 }
474 });
475
476 return GameResult<T>::ok(std::move(obj));
477 }

502 template <typename T>
503 [[nodiscard]] GameResult<T> deserializeJson(std::string_view json) const {
504 static_assert(detail::is_serializable_v<T>,
505 "Type must be registered with CGS_SERIALIZABLE");
506
507 detail::JsonReader reader{json};
508
509 if (!reader.expect('{')) {
510 return GameResult<T>::err(GameError(ErrorCode::InvalidJsonData, "expected '{'"));
511 }
512
513 T obj{};
514 auto fields = SerializableTraits<T>::fields();
515
516 bool first = true;
517 while (true) {
518 reader.skipWhitespace();
519 if (reader.pos >= reader.data.size()) {
520 return GameResult<T>::err(
521 GameError(ErrorCode::InvalidJsonData, "unexpected end of JSON"));
522 }
523 if (reader.data[reader.pos] == '}') {
524 ++reader.pos;
525 break;
526 }
527
528 if (!first) {
529 if (!reader.expect(',')) {
530 return GameResult<T>::err(
531 GameError(ErrorCode::InvalidJsonData, "expected ','"));
532 }
533 }
534 first = false;
535
536 // Read key
537 std::string key;
538 if (!reader.readQuotedString(key)) {
539 return GameResult<T>::err(
540 GameError(ErrorCode::InvalidJsonData, "expected key string"));
541 }
542 if (!reader.expect(':')) {
543 return GameResult<T>::err(GameError(ErrorCode::InvalidJsonData, "expected ':'"));
544 }
545
546 // Skip version field
547 if (key == "__v") {
548 reader.skipValue();
549 continue;
550 }
551
552 // Match against known fields
553 bool matched = false;
554 detail::forEachField(fields, [&](const auto& fd, std::size_t) {
555 if (matched)
556 return;
557 if (key != fd.name)
558 return;
559
560 using FieldType = std::remove_reference_t<decltype(obj.*(fd.pointer))>;
561 std::string rawVal;
562 if (reader.readRawValue(rawVal)) {
563 FieldType val{};
564 if (detail::parseJsonValue(rawVal, val)) {
565 obj.*(fd.pointer) = std::move(val);
566 }
567 matched = true;
568 }
569 });
570
571 if (!matched) {
572 reader.skipValue();
573 }
574 }
575
576 return GameResult<T>::ok(std::move(obj));
577 }
