55 [[nodiscard]] static inline ValidationResult validateEmail(std::string_view email) {
56 if (email.empty()) {
57 return ValidationResult::fail("email must not be empty");
58 }
59 if (email.size() > kMaxEmailLength) {
60 return ValidationResult::fail("email exceeds maximum length");
61 }
62
63 auto atPos = email.find('@');
64 if (atPos == std::string_view::npos || atPos == 0) {
65 return ValidationResult::fail("email must contain '@'");
66 }
67 // Only one '@' allowed.
68 if (email.find('@', atPos + 1) != std::string_view::npos) {
69 return ValidationResult::fail("email must contain exactly one '@'");
70 }
71
72 auto local = email.substr(0, atPos);
73 auto domain = email.substr(atPos + 1);
74
75 // -- Local part validation --
76 if (local.size() > kMaxLocalPartLength) {
77 return ValidationResult::fail("email local part exceeds 64 characters");
78 }
79 if (local.front() == '.' || local.back() == '.') {
80 return ValidationResult::fail("email local part must not start or end with '.'");
81 }
82 if (local.find("..") != std::string_view::npos) {
83 return ValidationResult::fail("email local part must not contain consecutive dots");
84 }
85 // RFC 5322 atext: alphanumeric + !#$%&'*+/=?^_`{|}~-.
86 for (char c : local) {
87 if (std::isalnum(static_cast<unsigned char>(c)))
88 continue;
89 if (isLocalSpecialChar(c))
90 continue;
91 return ValidationResult::fail("email local part contains invalid character");
92 }
93
94 // -- Domain validation --
95 if (domain.empty() || domain.size() > kMaxDomainLength) {
96 return ValidationResult::fail("email domain is invalid");
97 }
98 if (domain.front() == '.' || domain.back() == '.') {
99 return ValidationResult::fail("email domain must not start or end with '.'");
100 }
101 if (domain.find("..") != std::string_view::npos) {
102 return ValidationResult::fail("email domain must not contain consecutive dots");
103 }
104 // Must have at least one dot (TLD).
105 if (domain.find('.') == std::string_view::npos) {
106 return ValidationResult::fail("email domain must have at least one dot");
107 }
108
109 // Validate each label.
110 std::size_t labelStart = 0;
111 while (labelStart < domain.size()) {
112 auto dotPos = domain.find('.', labelStart);
113 auto labelEnd = (dotPos == std::string_view::npos) ? domain.size() : dotPos;
114 auto label = domain.substr(labelStart, labelEnd - labelStart);
115
116 if (label.empty() || label.size() > kMaxDomainLabelLength) {
117 return ValidationResult::fail("email domain label is invalid");
118 }
119 if (label.front() == '-' || label.back() == '-') {
120 return ValidationResult::fail("email domain label must not start or end with '-'");
121 }
122 for (char c : label) {
123 if (!std::isalnum(static_cast<unsigned char>(c)) && c != '-') {
124 return ValidationResult::fail("email domain contains invalid character");
125 }
126 }
127
128 labelStart = labelEnd + 1;
129 }
130
131 return ValidationResult::ok();
132 }

140 [[nodiscard]] static inline ValidationResult validatePassword(std::string_view password,
141 uint32_t minLength) {
142 if (password.size() < static_cast<std::size_t>(minLength)) {
143 return ValidationResult::fail("password must be at least " + std::to_string(minLength) +
144 " characters");
145 }
146 if (password.size() > kMaxPasswordLength) {
147 return ValidationResult::fail("password must not exceed " +
148 std::to_string(kMaxPasswordLength) + " characters");
149 }
150
151 bool hasUpper = false;
152 bool hasLower = false;
153 bool hasDigit = false;
154 bool hasSpecial = false;
155
156 for (char c : password) {
157 auto uc = static_cast<unsigned char>(c);
158 if (std::isupper(uc))
159 hasUpper = true;
160 else if (std::islower(uc))
161 hasLower = true;
162 else if (std::isdigit(uc))
163 hasDigit = true;
164 else
165 hasSpecial = true;
166 }
167
168 if (!hasUpper) {
169 return ValidationResult::fail("password must contain at least one uppercase letter");
170 }
171 if (!hasLower) {
172 return ValidationResult::fail("password must contain at least one lowercase letter");
173 }
174 if (!hasDigit) {
175 return ValidationResult::fail("password must contain at least one digit");
176 }
177 if (!hasSpecial) {
178 return ValidationResult::fail("password must contain at least one special character");
179 }
180
181 return ValidationResult::ok();

190 [[nodiscard]] static inline ValidationResult validateUsername(std::string_view username) {
191 if (username.size() < kMinUsernameLength) {
192 return ValidationResult::fail("username must be at least " +
193 std::to_string(kMinUsernameLength) + " characters");
194 }
195 if (username.size() > kMaxUsernameLength) {
196 return ValidationResult::fail("username must not exceed " +
197 std::to_string(kMaxUsernameLength) + " characters");
198 }
199
200 // Must start with a letter.
201 if (!std::isalpha(static_cast<unsigned char>(username.front()))) {
202 return ValidationResult::fail("username must start with a letter");
203 }
204
205 // Check allowed characters and consecutive specials.
206 bool prevSpecial = false;
207 for (char c : username) {
208 auto uc = static_cast<unsigned char>(c);
209 bool isSpecial = (c == '.' || c == '_' || c == '-');
210
211 if (!std::isalnum(uc) && !isSpecial) {
212 return ValidationResult::fail("username contains invalid character");
213 }
214 if (isSpecial && prevSpecial) {
215 return ValidationResult::fail(
216 "username must not contain consecutive special characters");
217 }
218 prevSpecial = isSpecial;
219 }
220
221 // Must not end with a special character.
222 if (username.back() == '.' || username.back() == '_' || username.back() == '-') {
223 return ValidationResult::fail("username must not end with a special character");
224 }
225
226 // Reserved names check (case-insensitive).
227 if (isReservedUsername(username)) {
228 return ValidationResult::fail("username is reserved");
229 }
230
231 return ValidationResult::ok();
232 }
