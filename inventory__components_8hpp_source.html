153struct Inventory {
154 std::vector<InventorySlot> slots;
155 uint32_t capacity = kDefaultInventoryCapacity;
156 int64_t currency = 0;
157
159 void Initialize() { slots.resize(capacity); }
160
169 uint32_t AddItem(const ItemTemplate& tmpl, uint32_t addCount) {
170 if (slots.empty()) {
171 Initialize();
172 }
173 uint32_t remaining = addCount;
174
175 // First pass: stack onto existing slots.
176 if (tmpl.IsStackable()) {
177 for (auto& slot : slots) {
178 if (remaining == 0) {
179 break;
180 }
181 if (slot.itemId == tmpl.id && slot.count < tmpl.maxStackSize) {
182 uint32_t space = tmpl.maxStackSize - slot.count;
183 uint32_t toAdd = std::min(remaining, space);
184 slot.count += toAdd;
185 remaining -= toAdd;
186 }
187 }
188 }
189
190 // Second pass: fill empty slots.
191 for (auto& slot : slots) {
192 if (remaining == 0) {
193 break;
194 }
195 if (slot.IsEmpty()) {
196 uint32_t toAdd = tmpl.IsStackable() ? std::min(remaining, tmpl.maxStackSize) : 1;
197 slot.itemId = tmpl.id;
198 slot.count = toAdd;
199 slot.durability = tmpl.maxDurability;
200 slot.maxDurability = tmpl.maxDurability;
201 remaining -= toAdd;
202 }
203 }
204
205 return addCount - remaining;
206 }
207
211 bool RemoveItem(uint32_t slotIndex, uint32_t removeCount) {
212 if (slotIndex >= slots.size()) {
213 return false;
214 }
215 auto& slot = slots[slotIndex];
216 if (slot.IsEmpty() || slot.count < removeCount) {
217 return false;
218 }
219
220 slot.count -= removeCount;
221 if (slot.count == 0) {
222 slot.Clear();
223 }
224 return true;
225 }
226
233 bool MoveItem(uint32_t fromSlot, uint32_t toSlot) {
234 if (fromSlot >= slots.size() || toSlot >= slots.size()) {
235 return false;
236 }
237 if (fromSlot == toSlot) {
238 return false;
239 }
240
241 auto& src = slots[fromSlot];
242 auto& dst = slots[toSlot];
243
244 if (src.IsEmpty()) {
245 return false;
246 }
247
248 // Stack merge if same item type.
249 if (dst.itemId == src.itemId && !dst.IsEmpty()) {
250 // Caller must know max stack size externally for proper merge.
251 // For now, swap if they can't trivially merge.
252 std::swap(src, dst);
253 return true;
254 }
255
256 std::swap(src, dst);
257 return true;
258 }
259
265 std::optional<uint32_t> SplitStack(uint32_t slotIndex, uint32_t splitCount) {
266 if (slotIndex >= slots.size()) {
267 return std::nullopt;
268 }
269 auto& src = slots[slotIndex];
270 if (src.IsEmpty() || src.count <= splitCount || splitCount == 0) {
271 return std::nullopt;
272 }
273
274 // Find an empty slot.
275 for (uint32_t i = 0; i < static_cast<uint32_t>(slots.size()); ++i) {
276 if (slots[i].IsEmpty()) {
277 slots[i].itemId = src.itemId;
278 slots[i].count = splitCount;
279 slots[i].durability = src.durability;
280 slots[i].maxDurability = src.maxDurability;
281 src.count -= splitCount;
282 return i;
283 }
284 }
285 return std::nullopt;
286 }

335struct Equipment {

341 InventorySlot Equip(EquipSlot slot, InventorySlot item) {
342 auto idx = static_cast<std::size_t>(slot);
343 if (idx >= kEquipSlotCount) {
344 return {};
345 }
346
347 InventorySlot previous = std::move(slots[idx]);
348 item.count = 1; // Equipment slots always hold 1 item.
349 slots[idx] = std::move(item);
350 return previous;
351 }
352
356 InventorySlot Unequip(EquipSlot slot) {
357 auto idx = static_cast<std::size_t>(slot);
358 if (idx >= kEquipSlotCount) {
359 return {};
360 }
361
362 InventorySlot removed = std::move(slots[idx]);
363 slots[idx] = InventorySlot{};
364 return removed;
365 }
366
368 [[nodiscard]] const InventorySlot* GetEquipped(EquipSlot slot) const {
369 auto idx = static_cast<std::size_t>(slot);
370 if (idx >= kEquipSlotCount) {
371 return nullptr;
372 }
373 return slots[idx].IsEmpty() ? nullptr : &slots[idx];
374 }
375
382 [[nodiscard]] StatBonuses CalculateStatBonuses(
383 const std::vector<ItemTemplate>& templates) const {
384 StatBonuses total;
385 for (const auto& slot : slots) {
386 if (slot.IsEmpty() || slot.IsBroken()) {
387 continue;
388 }
389
390 // Look up the item template for base stat bonuses.
391 for (const auto& tmpl : templates) {
392 if (tmpl.id == slot.itemId) {
393 total += tmpl.statBonuses;
394 break;
395 }
396 }
397
398 // Add enchantment bonuses.
399 total += slot.GetEnchantBonuses();
400 }
401 return total;
402 }
