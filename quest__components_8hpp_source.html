28struct QuestObjective {
29 ObjectiveType type = ObjectiveType::Kill;
30 uint32_t targetId = 0;
31 int32_t current = 0;
32 int32_t required = 1;
33 bool completed = false;
34
36 void AddProgress(int32_t amount) noexcept {
37 if (completed) {
38 return;
39 }
40 current = std::min(current + amount, required);
41 if (current >= required) {
42 completed = true;
43 }
44 }
45
47 [[nodiscard]] bool IsComplete() const noexcept { return completed || current >= required; }
48};

78struct QuestEntry {
79 uint32_t questId = 0;
80 uint32_t templateId = 0;
81 QuestState state = QuestState::Accepted;
82 std::vector<QuestObjective> objectives;
83 float elapsedTime = 0.0f;
84 float timeLimit = 0.0f;
85
87 [[nodiscard]] bool AllObjectivesComplete() const {
88 return std::all_of(objectives.begin(), objectives.end(), [](const QuestObjective& obj) {
89 return obj.IsComplete();
90 });
91 }
92
96 bool UpdateObjective(ObjectiveType type, uint32_t targetId, int32_t amount = 1) {
97 if (state != QuestState::Accepted) {
98 return false;
99 }
100 bool updated = false;
101 for (auto& obj : objectives) {
102 if (obj.type == type && obj.targetId == targetId && !obj.completed) {
103 obj.AddProgress(amount);
104 updated = true;
105 }
106 }
107 if (updated && AllObjectivesComplete()) {
108 state = QuestState::ObjectivesComplete;
109 }
110 return updated;
111 }
112};

120struct QuestLog {

128 bool Accept(const QuestTemplate& tmpl) {
129 if (activeQuests.size() >= maxActiveQuests) {
130 return false;
131 }
132 if (HasQuest(tmpl.id)) {
133 return false;
134 }
135 if (!CanAccept(tmpl)) {
136 return false;
137 }
138
139 QuestEntry entry;
140 entry.questId = tmpl.id;
141 entry.templateId = tmpl.id;
142 entry.state = QuestState::Accepted;
143 entry.objectives = tmpl.objectives;
144 entry.timeLimit = tmpl.timeLimitSeconds;
145 activeQuests.push_back(std::move(entry));
146 return true;
147 }
148
152 bool Abandon(uint32_t questId) {
153 auto it = std::find_if(activeQuests.begin(),
154 activeQuests.end(),
155 [questId](const QuestEntry& e) { return e.questId == questId; });
156 if (it == activeQuests.end()) {
157 return false;
158 }
159 activeQuests.erase(it);
160 return true;
161 }
162
166 bool TurnIn(uint32_t questId) {
167 auto* entry = GetQuest(questId);
168 if (!entry || entry->state != QuestState::ObjectivesComplete) {
169 return false;
170 }
171 entry->state = QuestState::TurnedIn;
172 completedQuestIds.insert(questId);
173 return true;
174 }

177 [[nodiscard]] QuestEntry* GetQuest(uint32_t questId) {

181 }
182 }
183 return nullptr;
184 }

188 for (const auto& entry : activeQuests) {

194 }

197 [[nodiscard]] bool HasQuest(uint32_t questId) const {
198 return std::any_of(activeQuests.begin(),

201 }
202
204 [[nodiscard]] bool IsCompleted(uint32_t questId) const {

209 [[nodiscard]] bool CanAccept(const QuestTemplate& tmpl) const {
210 if (HasQuest(tmpl.id)) {
211 return false;
212 }
213 if (IsCompleted(tmpl.id) && !HasQuestFlag(tmpl.flags, QuestFlags::Repeatable)) {
214 return false;
215 }
216 return std::all_of(tmpl.prerequisites.begin(),
217 tmpl.prerequisites.end(),
218 [this](uint32_t prereqId) { return IsCompleted(prereqId); });
219 }

225 });
