390template <typename T, typename... Args>
391T& SystemScheduler::Register(Args&&... args) {
392 static_assert(std::is_base_of_v<ISystem, T>, "T must derive from ISystem");
393
394 const auto typeId = SystemType<T>::Id();
395
396 // Return existing instance if already registered.
397 if (auto it = systems_.find(typeId); it != systems_.end()) {
398 return static_cast<T&>(*it->second.instance);
399 }
400
401 auto system = std::make_unique<T>(std::forward<Args>(args)...);
402 T& ref = *system;
403
404 SystemEntry entry;
405 entry.instance = std::move(system);
406 entry.typeId = typeId;
407 entry.stage = ref.GetStage();
408 entry.enabled = true;
409
410 stageGroups_[entry.stage].push_back(typeId);
411 systems_.emplace(typeId, std::move(entry));
412
413 // Invalidate any previously built plan.
414 built_ = false;
415
416 return ref;
417}
